1.辗转相除法
ll gcd(ll a,ll b)
{
  if(!b)return a;
  return gcd(b,a % b);
}
2.
ll exgcd(ll a,ll b,ll &x,ll &y)
{
  if(!b)
  {
    x = 1,y = 0;
    return a;
  }
  int d = exgcd(b,a % b,y,x);
  y -= (a / b) * x;
  return d;
}
//保证a,b为正，为负变形一下就行，所求的便是-x
//传回的x，y为一组特解
